    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        //Edit Button
        try{
            String sql = "update karyawan set bpjs=?,anak=?,transport=?,makan=? where id_tunjangan='"+idtunjangan.getText()+"'";
            PreparedStatement stat = conn.prepareStatement(sql);
            stat.setString(1, bpjs.getText());
            stat.setString(2, anak.getText());
            stat.setString(3, transport.getText());
            stat.setString(4, makan.getText());
            stat.executeUpdate();
            JOptionPane.showMessageDialog(null, "Edit successful!");
            kosong();
            idtunjangan.requestFocus();
        }
        catch (SQLException e){
            JOptionPane.showMessageDialog(null, "Failed to change"+e);
        }
        datatable();
    }                                        

    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        //Delete Button
        int ok = JOptionPane.showConfirmDialog(null,"hapus","konfirmasi dialog",JOptionPane.YES_NO_OPTION);
        if (ok==0){
            String sql = "delete from tunjangan where id_tunjangan ='"+idtunjangan.getText()+"'";
            try{
                PreparedStatement stat = conn.prepareStatement(sql);
                stat.executeUpdate();
                JOptionPane.showMessageDialog(null, "Delete succesfull!");
                kosong();
                idtunjangan.requestFocus();
            }
            catch (SQLException e){
                JOptionPane.showMessageDialog(null, "Failed to delete "+e);
            }
            datatable();
        }
    }

        // TODO add your handling code here:
        // Submit button
        try{
            String sql = "INSERT INTO tunjangan values (?,?,?,?,?)";
            PreparedStatement stat = conn.prepareStatement(sql);
            stat.setString(1, idtunjangan.getText());
            stat.setString(2, bpjs.getText());
            stat.setString(3, anak.getText());
            stat.setString(4, transport.getText());
            stat.setString(5, makan.getText());
